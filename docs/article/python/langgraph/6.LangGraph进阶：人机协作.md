# LangGraph 进阶：人机协作

在实际应用中，我们往往不希望 Agent 完全自主决策——某些操作（如发送邮件、执行数据库变更、支付操作）需要人工确认。LangGraph 提供了 **interrupt（中断）** 机制来实现人机协作。

## 中断机制概述

LangGraph 的中断允许你在图执行到指定节点时**暂停**，等待人工输入后再**恢复**执行：

```
[START] → [agent] → [中断：等待人工确认] → [execute] → [END]
```

### 前提条件

中断需要 **Checkpointer** 支持（必须启用持久化），因为中断就是在检查点处暂停。

## interrupt_before 和 interrupt_after

编译图时指定在哪些节点前/后中断：

```python
from langgraph.checkpoint.memory import MemorySaver

memory = MemorySaver()

graph = graph_builder.compile(
    checkpointer=memory,
    interrupt_before=["execute"],  # 执行 execute 节点之前中断
)
```

### interrupt_before

在目标节点**执行之前**暂停。适合在执行危险操作前要求确认：

```python
graph = graph_builder.compile(
    checkpointer=memory,
    interrupt_before=["send_email", "delete_record"]
)
```

### interrupt_after

在目标节点**执行之后**暂停。适合让人审核节点的输出结果：

```python
graph = graph_builder.compile(
    checkpointer=memory,
    interrupt_after=["draft_email"]  # 审核邮件草稿
)
```

## 基础示例：工具调用审批

构建一个需要人工批准工具调用的 Agent：

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages
from langgraph.prebuilt import ToolNode
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool
from dotenv import load_dotenv

load_dotenv()

class State(TypedDict):
    messages: Annotated[list, add_messages]

@tool
def send_email(to: str, subject: str, body: str) -> str:
    """发送邮件"""
    return f"邮件已发送给 {to}，主题：{subject}"

@tool
def search(query: str) -> str:
    """搜索信息"""
    return f"搜索 '{query}' 的结果：..."

tools = [send_email, search]
llm = ChatOpenAI(model="gpt-4o-mini").bind_tools(tools)

def agent(state: State):
    response = llm.invoke(state["messages"])
    return {"messages": [response]}

def should_continue(state: State) -> str:
    last_message = state["messages"][-1]
    if hasattr(last_message, "tool_calls") and last_message.tool_calls:
        return "tools"
    return END

graph_builder = StateGraph(State)
graph_builder.add_node("agent", agent)
graph_builder.add_node("tools", ToolNode(tools))
graph_builder.add_edge(START, "agent")
graph_builder.add_conditional_edges("agent", should_continue)
graph_builder.add_edge("tools", "agent")

memory = MemorySaver()

# 在工具节点执行前中断
graph = graph_builder.compile(
    checkpointer=memory,
    interrupt_before=["tools"]
)
```

### 交互流程

```python
config = {"configurable": {"thread_id": "email-demo"}}

# 第一步：用户请求
result = graph.invoke(
    {"messages": [{"role": "user", "content": "帮我给 boss@company.com 发一封邮件，主题是请假申请"}]},
    config
)

# 图在 tools 节点前中断
# 查看 Agent 准备调用的工具
state = graph.get_state(config)
last_msg = state.values["messages"][-1]
print(f"Agent 准备调用: {last_msg.tool_calls}")
# [{'name': 'send_email', 'args': {'to': 'boss@company.com', ...}}]

# 人工决策
user_approval = input("是否批准发送邮件？(y/n): ")

if user_approval.lower() == "y":
    # 批准 → 继续执行（传 None 表示从中断点恢复）
    result = graph.invoke(None, config)
    print(f"结果: {result['messages'][-1].content}")
else:
    # 拒绝 → 手动注入一条拒绝消息
    from langchain_core.messages import ToolMessage

    tool_call = last_msg.tool_calls[0]
    graph.update_state(
        config,
        {
            "messages": [
                ToolMessage(
                    content="操作已被用户取消",
                    tool_call_id=tool_call["id"]
                )
            ]
        }
    )
    result = graph.invoke(None, config)
    print(f"结果: {result['messages'][-1].content}")
```

## 使用 interrupt() 函数

除了在编译时指定中断节点，你还可以在节点内部使用 `interrupt()` 函数实现更灵活的中断：

```python
from langgraph.types import interrupt, Command

def review_node(state: State):
    """需要人工审核的节点"""
    draft = state["draft"]

    # 中断并向用户展示信息
    decision = interrupt({
        "type": "review",
        "draft": draft,
        "prompt": "请审核以上草稿，输入 approve 批准或 reject 拒绝"
    })

    # 人工输入后继续
    if decision == "approve":
        return {"status": "approved", "final_content": draft}
    else:
        return {"status": "rejected"}
```

恢复中断：

```python
# 图在 interrupt() 处暂停
state = graph.get_state(config)

# 使用 Command 恢复，传入人工决策
graph.invoke(Command(resume="approve"), config)
```

## 动态修改状态

`update_state` 不仅可以添加数据，还可以用于修正 Agent 的决策：

```python
# Agent 准备调用错误的工具，人工修正
state = graph.get_state(config)
last_msg = state.values["messages"][-1]

# 修改工具调用参数
corrected_tool_calls = last_msg.tool_calls.copy()
corrected_tool_calls[0]["args"]["to"] = "correct@email.com"

from langchain_core.messages import AIMessage

corrected_msg = AIMessage(
    content=last_msg.content,
    tool_calls=corrected_tool_calls,
    id=last_msg.id  # 使用相同 ID 会替换原消息
)

graph.update_state(config, {"messages": [corrected_msg]})

# 用修正后的参数继续执行
result = graph.invoke(None, config)
```

## 实战：多步审批流程

一个需要多级审批的工作流：

```python
from typing import Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.memory import MemorySaver
from langgraph.types import interrupt

class ApprovalState(TypedDict):
    request: str
    amount: float
    manager_approved: bool
    director_approved: bool
    result: str

def submit_request(state: ApprovalState):
    return {"result": f"申请已提交：{state['request']}，金额：{state['amount']}"}

def manager_review(state: ApprovalState):
    decision = interrupt({
        "level": "经理审批",
        "request": state["request"],
        "amount": state["amount"]
    })
    return {"manager_approved": decision == "approve"}

def director_review(state: ApprovalState):
    decision = interrupt({
        "level": "总监审批",
        "request": state["request"],
        "amount": state["amount"]
    })
    return {"director_approved": decision == "approve"}

def route_after_manager(state: ApprovalState) -> str:
    if not state["manager_approved"]:
        return "rejected"
    if state["amount"] > 10000:
        return "director_review"
    return "approved"

def approved(state: ApprovalState):
    return {"result": "✅ 申请已通过"}

def rejected(state: ApprovalState):
    return {"result": "❌ 申请已被拒绝"}

graph_builder = StateGraph(ApprovalState)
graph_builder.add_node("submit", submit_request)
graph_builder.add_node("manager_review", manager_review)
graph_builder.add_node("director_review", director_review)
graph_builder.add_node("approved", approved)
graph_builder.add_node("rejected", rejected)

graph_builder.add_edge(START, "submit")
graph_builder.add_edge("submit", "manager_review")
graph_builder.add_conditional_edges("manager_review", route_after_manager)
graph_builder.add_edge("director_review", "approved")  # 总监审批后直接通过（简化示例）
graph_builder.add_edge("approved", END)
graph_builder.add_edge("rejected", END)

memory = MemorySaver()
graph = graph_builder.compile(checkpointer=memory)
```

```
[提交] → [经理审批] ──批准──→ 金额>1万 → [总监审批] → [通过]
                     │                              
                     └──拒绝──→ [拒绝]
                     │
                     └──批准且≤1万──→ [通过]
```

## 小结

| 概念 | 说明 |
|------|------|
| `interrupt_before` | 在指定节点执行前暂停 |
| `interrupt_after` | 在指定节点执行后暂停 |
| `interrupt()` | 在节点内部触发中断 |
| `Command(resume=...)` | 恢复中断并传入用户输入 |
| `update_state()` | 手动修改图的状态 |
| `invoke(None, config)` | 从中断点恢复执行 |
